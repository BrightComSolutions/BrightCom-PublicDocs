{{/* BrightCom Docs — card shortcode override. Emits a bc-product-card matching the home grid aesthetic. Named params (backward-compat with Docsy's card): header = product name (markdown bold like "**BRC Connect**" OK) title = category / secondary label mark = optional 2-letter override (e.g. "BC"); derived if omitted url = optional explicit link; otherwise the first href in inner content wins The trailing "

Learn More" pattern is stripped from the rendered description since the whole card becomes the link. */}} {{- $inner := trim .Inner " \n\r\t" -}} {{- $header := .Get "header" | default "" -}} {{- $title := .Get "title" | default "" -}} {{- $markParam := .Get "mark" | default "" -}} {{- $urlParam := .Get "url" | default "" -}} {{- /* Strip markdown emphasis from header */ -}} {{- $name := replaceRE `\*\*|__|\*|_` "" $header -}} {{- $name = trim $name " " -}} {{- /* Resolve URL: explicit param beats first in inner */ -}} {{- $url := $urlParam -}} {{- if eq $url "" -}} {{- $hrefMatches := findRESubmatch `href=["']([^"']+)["']` $inner -}} {{- if $hrefMatches -}} {{- $url = index (index $hrefMatches 0) 1 -}} {{- end -}} {{- end -}} {{- if eq $url "" -}}{{- $url = "#" -}}{{- end -}} {{- /* Derive the 2-letter mark if not passed in. Strip a leading "BRC " (or "BRC") then take the first two letters of the remainder. */ -}} {{- $mark := $markParam -}} {{- if eq $mark "" -}} {{- $plain := $name -}} {{- $plain = replaceRE `^(?i)BRC\s+` "" $plain -}} {{- $plain = replaceRE `[^A-Za-z]` "" $plain -}} {{- if ge (len $plain) 2 -}} {{- $mark = upper (substr $plain 0 2) -}} {{- else -}} {{- $mark = upper (substr (replaceRE `[^A-Za-z]` "" $name) 0 2) -}} {{- end -}} {{- end -}} {{- /* Strip the trailing "

Learn More" junk for a clean description — the card itself becomes the link. */ -}} {{- $desc := $inner -}} {{- $desc = replaceRE `(?is)\s*([\s\n]*)*\s*]*>[^<]*\s*$` "" $desc -}} {{- $desc = trim $desc " \n\r\t" -}}
